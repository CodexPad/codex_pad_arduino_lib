116void loop() {
117 // 重要：Update()方法必须在循环中尽可能频繁地调用，不能添加延时
118 // 该方法负责处理所有接收到的蓝牙数据包，延时会导致数据丢失和响应延迟
119 // 对于实时控制应用，必须保持高频率调用以确保及时响应手柄输入
120 // Important: Update() method must be called as frequently as possible in the loop, no delays should be added
121 // This method processes all received Bluetooth packets, delays will cause data loss and response lag
122 // For real-time control applications, high-frequency calls are essential to ensure prompt response to gamepad input
123 g_codex_pad.Update();
124
125 if (!g_codex_pad.is_connected()) {
126 printf("disconnected, reconnecting...\n");
127 const auto ret = g_codex_pad.Connect(kMacAddress);
128 printf("connected, ret: %d\n", ret);
129 }
130
131 // 检测所有按钮的状态变化
132 // 使用pressed(), released(), holding()方法检测按钮的不同状态
133 // Detect state changes for all buttons
134 // Use pressed(), released(), holding() methods to detect different button states
135 for (auto button : {CodexPad::Button::kUp,
136 CodexPad::Button::kDown,
137 CodexPad::Button::kLeft,
138 CodexPad::Button::kRight,
139 CodexPad::Button::kSquareX,
140 CodexPad::Button::kTriangleY,
141 CodexPad::Button::kCrossA,
142 CodexPad::Button::kCircleB,
143 CodexPad::Button::kL1,
144 CodexPad::Button::kL2,
145 CodexPad::Button::kL3,
146 CodexPad::Button::kR1,
147 CodexPad::Button::kR2,
148 CodexPad::Button::kR3,
149 CodexPad::Button::kSelect,
150 CodexPad::Button::kStart,
151 CodexPad::Button::kHome}) {
152 // 检测按钮是否刚刚按下（从弹起变为按下）
153 // Check if button was just pressed (transition from released to pressed)
154 if (g_codex_pad.pressed(button)) {
155 printf("Button %s: pressed\n", ButtonToString(button).c_str());
156 }
157 // 检测按钮是否刚刚释放（从按下变为弹起）
158 // Check if button was just released (transition from pressed to released)
159 else if (g_codex_pad.released(button)) {
160 printf("Button %s: released\n", ButtonToString(button).c_str());
161 }
162 // 检测按钮是否持续按下状态
163 // Check if button is holding
164 else if (g_codex_pad.holding(button)) {
165 printf("Button %s: holding\n", ButtonToString(button).c_str());
166 }
167 }
168
169 // 检测摇杆轴值是否发生了有效变化（使用阈值避免微小抖动）
170 // 阈值设置为2，只有当摇杆值变化达到或超过2个单位时才认为是有效变化
171 // Check if joystick axis values have changed significantly (using threshold to avoid minor jitter)
172 // Threshold is set to 2, only consider changes equal to or greater than 2 units as significant
173 constexpr uint8_t kAxisValueChangeThreshold = 2;
174
175 // 检测左摇杆X轴或Y轴是否有显著变化
176 // Check if left stick X or Y axis has significant change
177 if (g_codex_pad.HasAxisValueChanged(CodexPad::Axis::kLeftStickX, kAxisValueChangeThreshold) ||
178 g_codex_pad.HasAxisValueChanged(CodexPad::Axis::kLeftStickY, kAxisValueChangeThreshold) ||
179 g_codex_pad.HasAxisValueChanged(CodexPad::Axis::kRightStickX, kAxisValueChangeThreshold) ||
180 g_codex_pad.HasAxisValueChanged(CodexPad::Axis::kRightStickY, kAxisValueChangeThreshold)) {
181 // 打印摇杆轴的当前值（0-255）
182 // Print current joystick axis values (0-255)
183 printf("L(%3" PRIu8 ", %3" PRIu8 "), R(%3" PRIu8 ", %3" PRIu8 ")\n",
184 g_codex_pad.axis_value(CodexPad::Axis::kLeftStickX), // 左摇杆X轴当前值 | Left stick X axis current value
185 g_codex_pad.axis_value(CodexPad::Axis::kLeftStickY), // 左摇杆Y轴当前值 | Left stick Y axis current value
186 g_codex_pad.axis_value(CodexPad::Axis::kRightStickX), // 右摇杆X轴当前值 | Right stick X axis current value
187 g_codex_pad.axis_value(CodexPad::Axis::kRightStickY) // 右摇杆Y轴当前值 | Right stick Y axis current value
188 );
189 }
